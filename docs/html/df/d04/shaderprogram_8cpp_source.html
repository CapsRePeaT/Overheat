92bool ShaderProgram::Load(const Path& vertex_path, const Path& fragment_path) {
93 // TODO: find out if we need explicitly check errors with debug callback
94 // Create vertex and fragment shaders
95 uint32_t vs;
96 uint32_t fs;
97 bool shaders_compiled = true;
98 shaders_compiled &= CreateShader(vertex_path, GL_VERTEX_SHADER, vs);
99 shaders_compiled &= CreateShader(fragment_path, GL_FRAGMENT_SHADER, fs);
100 if (!shaders_compiled)
101 return is_compiled_ = false;
102 // Create shader program, link and delete the shaders
103 id_ = glCreateProgram();
104 glAttachShader(id_, vs);
105 glAttachShader(id_, fs);
106 glLinkProgram(id_);
107 glDeleteShader(vs);
108 glDeleteShader(fs);
109 return is_compiled_ = CheckSuccessAndLogError<GL_LINK_STATUS>(
110 id_, "SHADER: Link time error:");
111}

113bool CreateShader(const ShaderProgram::Path& file_path,
114 const GLenum shader_type, uint32_t& shader_id) {
115 // Open and read file into code_pointer
116 std::ifstream file_stream;
117 file_stream.open(file_path, std::ios::in | std::ios::binary);
118 if (!file_stream.is_open()) {
119 spdlog::error("SHADER: Failed to open file: {}", file_path.string());
120 return false;
121 }
122 std::stringstream buff;
123 buff << file_stream.rdbuf();
124 const auto code_string = buff.str();
125 const std::array<const char*, 1> source_codes = {code_string.c_str()};
126 // Compile read code
127 shader_id = glCreateShader(shader_type);
128 glShaderSource(shader_id, source_codes.size(), source_codes.data(),
129 /*length=*/nullptr);
130 glCompileShader(shader_id);
131 bool success = CheckSuccessAndLogError<GL_COMPILE_STATUS>(
132 shader_id, "SHADER: Compile time error in {}", file_path.string());
133 if (!success)
134 glDeleteShader(shader_id);
135 return success;
136}
