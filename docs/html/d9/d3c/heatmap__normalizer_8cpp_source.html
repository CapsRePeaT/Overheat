34Heatmap HeatmapNormalizer::BilinearInterpolateSlow(const Heatmap& heatmap) {
35 Floats result;
36 result.reserve(resolution_ * resolution_);
37 for (size_t i = 0; i < resolution_; ++i) {
38 for (size_t j = 0; j < resolution_; ++j) {
39 float x_coord = j * x_new_step_;
40 float y_coord = i * y_new_step_;
41 auto [x_bounds, y_bounds, upper_values, lower_values] =
42 FindBilinearInterpolationCell(heatmap, x_coords_, y_coords_,
43 {x_coord, y_coord}, env_temp_);
44 // r1
45 // |--------------|
46 // | | |
47 // | | |
48 // | | |
49 // |-----result---| |--->
50 // | | | | x
51 // | | | |
52 // |--------------| ∨ y
53 // r2
54 auto r1 = Lerp(x_bounds, upper_values, x_coord);
55 auto r2 = Lerp(x_bounds, lower_values, x_coord);
56 result.push_back(Lerp(y_bounds, {r1, r2}, y_coord));
57 }
58 }
59 return Heatmap(std::move(result), resolution_, resolution_);
60}

62HeatmapNormalizer::Cell HeatmapNormalizer::FindBilinearInterpolationCell(
63 const Heatmap& heatmap, const Floats& x_coords, const Floats& y_coords,
64 FloatPair search_coords, float env_temp) {
65 namespace ranges = std::ranges;
66
67 assert(search_coords.first >= x_coords.front());
68 assert(search_coords.second >= y_coords.front());
69 // upper_bound searches for an element, that is greater than given value. So
70 // we always get [begin() + 1, end()) from it here, that is good as
71 // right/lower side of cell.
72 auto lower_coord_it = ranges::upper_bound(y_coords, search_coords.second);
73 if (lower_coord_it == y_coords.end())
74 lower_coord_it = y_coords.end() - 1;
75 auto upper_coord_it = lower_coord_it - 1;
76
77 auto right_coord_it = ranges::upper_bound(x_coords, search_coords.first);
78 if (right_coord_it == x_coords.end())
79 right_coord_it = x_coords.end() - 1;
80 auto left_coord_it = right_coord_it - 1;
81
82 auto upper_row_idx = std::distance(y_coords.begin(), upper_coord_it);
83 auto lower_row_idx = std::distance(y_coords.begin(), lower_coord_it);
84 auto left_column_idx = std::distance(x_coords.begin(), left_coord_it);
85 auto right_column_idx = std::distance(x_coords.begin(), right_coord_it);
86
87 assert(upper_row_idx == lower_row_idx - 1);
88 assert(left_column_idx == right_column_idx - 1);
89
90 auto upper_row = heatmap.row(upper_row_idx);
91 auto lower_row = heatmap.row(lower_row_idx);
92
93 FloatPair upper_values = {upper_row[left_column_idx],
94 upper_row[right_column_idx]};
95 FloatPair lower_values = {lower_row[left_column_idx],
96 lower_row[right_column_idx]};
97
98 return {
99 .x_bounds = {*left_coord_it, *right_coord_it},
100 .y_bounds = {*upper_coord_it, *lower_coord_it},
101 .upper_values = upper_values,
102 .lower_values = lower_values,
103 };
104}

171// Normalizes heatmap values to range [0; 1] without lossing min/max temperature
172// information
173void HeatmapNormalizer::Normalize(Heatmap& heatmap) {
174 auto min_max_diff = heatmap.max_temp() - heatmap.min_temp();
175 for (size_t i = 0; i < heatmap.y_resolution(); ++i) {
176 auto row = heatmap.row(i);
177 std::ranges::transform(row, row.begin(),
178 [&heatmap, min_max_diff](auto temp) {
179 return (temp - heatmap.min_temp()) / min_max_diff;
180 });
181 }
182}

184size_t HeatmapNormalizer::FindOptimalResolution(const float min_old_step,
185 const float max_dimension_size,
186 const size_t max_resolution) {
187 assert(max_resolution > 0);
188 const auto old_max_resolution =
189 static_cast<size_t>(max_dimension_size / min_old_step) + 1;
190
191 // Minimal meanfull resolution for real heatmaps.
192 static constexpr size_t min_resolution = 512;
193 // Max resolution should be passed from renderer api restriction, but may be
194 // less than min_resolution for tests and should be using in this case.
195 size_t heatmap_resolution = std::min(min_resolution, max_resolution);
196 while (heatmap_resolution < old_max_resolution &&
197 heatmap_resolution < max_resolution) {
198 heatmap_resolution *= 2;
199 }
200
201 //assert(max_dimension_size / heatmap_resolution < min_old_step);
202
203 return std::min(heatmap_resolution, max_resolution);
204}

206float HeatmapNormalizer::Lerp(const std::pair<float, float> bound_coords,
207 const std::pair<float, float> values,
208 const float interpolation_coord) {
209 const float overall_distance = bound_coords.second - bound_coords.first;
210
211 // assert(std::abs(overall_distance) > eps);
212 if (std::abs(overall_distance) < eps)
213 return values.second;
214 assert(interpolation_coord > bound_coords.first ||
215 float_eq(interpolation_coord, bound_coords.first));
216
217 const float factor =
218 (interpolation_coord - bound_coords.first) / overall_distance;
219 return std::lerp(values.first, values.second, factor);
220}

222size_t HeatmapNormalizer::CalculateBatchSize(const float right_coord,
223 const float start_coord,
224 const float step_size) {
225 assert(std::abs(step_size) > eps);
226
227 // step_size = 1
228 //
229 // 1 2 3
230 // |-----------|-----------|
231 // start_coord right_coord
232 //
233 // In this case function may return 2 or 3 due to float roundness.
234 // So we check if we got result, which may include right_coord with
235 // incrementing, we are incrementing it.
236 //
237 size_t res = 1 + static_cast<size_t>((right_coord - start_coord) / step_size);
238 if (std::abs((res + 1) * step_size - right_coord) < eps)
239 ++res;
240 return res;
241}
