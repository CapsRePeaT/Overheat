25RendererWidget::~RendererWidget() {
26 // Context destroying is guaranted to happen after OpenGLWidget destruction
27 // (see https://doc.qt.io/qt-6/qopenglwidget.html#dtor.QOpenGLWidget).
28 // Hence, in order to avoid SEGFAULT on signal invoking (because it connected
29 // to *this) we must disconnect the signal.
30 // Also we can explicitly reset the pointer, or call ClearResources, but it
31 // will be done anyway as a part of RendererWidget destruction
32 disconnect(context(), &QOpenGLContext::aboutToBeDestroyed, this, nullptr);
33 spdlog::debug("Render widget destroyed");
34}

36void RendererWidget::initializeGL() {
37 // We need know which scene to link to viewport after creating RendererWidget,
38 // hence we have to store it in class.
39 // As RendererWidget is considered to be owned by somthing, that also owns
40 // Scene, we can store weak_ptr here.
41 assert(!scene_.expired() && "Scene was destroyed before RendererWidget");
42 auto&& scene = scene_.lock();
43 // TODO: reuse the pointer
44 viewport_ =
45 renderer::ISceneViewport::Create(renderer::API::OpenGL, std::move(scene));
46 viewport_->Initialize(width(), height());
47 // Gracefully clear resources of renderer when context is destroying
48 // This is needed in cases, when we manually destroying context in a
49 // lifetime of one RenderWidget (that is unlikely, but still).
50 // For additional information see
51 // https://doc.qt.io/qt-6/qopenglcontext.html#aboutToBeDestroyed
52 //
53 // Also this can be done with renderer_.reset(), because:
54 // 1. its resource cleaning must be called in dtors of IRenderer
55 // subclasses too;
56 // 2. on next context initializing we reassign unique_ptr anyway
57 // (see 1st line of this function).
58 // But 1. I want to reuse the pointer in near future (TBT) and
59 // 2. this is more self-documented
60 connect(context(), &QOpenGLContext::aboutToBeDestroyed, this,
61 [&renderer = *viewport_]() { renderer.ClearResources(); });
62}
