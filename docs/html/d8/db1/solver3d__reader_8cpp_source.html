70Solver3d_TRM read_geometry(const std::string& content) {
71 // splits layers for groups (inside/outside box)
72 const auto groups =

74
75 if (groups.size() < 2)

79 read_general_info(content, data);

81 // read layers

83 std::transform(
84 groups.begin(), groups.end(),

88 const auto layers_tags = split(group, layers_regex);
