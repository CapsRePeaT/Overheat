114void GLSceneViewport::RenderFrame() {
115 auto& api = RendererAPI::instance();
116 api.Clear();
117
118 const ICamera& camera = camera_controller_->camera();
119
120 if (!heatmap_materials_ && !scene_->heatmaps().empty()) {
121 InitHeatmapMaterials();
122 }
123
124 if (heatmap_materials_)
125 for (const auto& shape : scene_->shapes()) {
126 // LOG_TRACE("Render shape: id {}, layer {}", shape->id().id(),
127 // shape->layer_id());
128 if (shape->is_visible() ||
129 shape->highlight_type() != HighlightType::None) {
130 (*heatmap_materials_)[shape->core_shape().layer_id()].Use(
131 shape->transform(), camera.viewProjectionMatrix(),
132 HighlightTypeToColor(shape->highlight_type()),
133 !shape->is_visible());
134 api.DrawIndexed(shape->vertex_array());
135 }
136 }
137
138 if (data_) {
139 const auto& axes = data_->axes;
140 data_->debug_material->Use(axes->transform(), camera.viewProjectionMatrix(),
141 1.0f);
142 api.DrawIndexed(axes->vertex_array(), PrimitiveType::LINES);
143 }

146HeatmapMaterial CreateMaterial(const Heatmap& bot_heatmap,
147 const Heatmap& top_heatmap,
148 std::pair<float, float> bounds) {
149 auto& factory = RendererAPI::factory();
150 int side_resolution = bot_heatmap.x_resolution();
151 auto heatmap_bottom_texture = factory.NewTexture2D(
152 side_resolution, side_resolution, bot_heatmap.temperatures().data(), 1);
153 auto heatmap_top_texture = factory.NewTexture2D(
154 side_resolution, side_resolution, top_heatmap.temperatures().data(), 1);
155
156 auto texture_pair =
157 std::pair<std::unique_ptr<Texture2D>, std::unique_ptr<Texture2D>>(
158 std::move(heatmap_bottom_texture), std::move(heatmap_top_texture));
159 auto temp_ranges_pair = std::pair<glm::vec2, glm::vec2>(
160 {bot_heatmap.min_temp(), bot_heatmap.max_temp()},
161 {top_heatmap.min_temp(), top_heatmap.max_temp()});
162
163 return {std::move(texture_pair), temp_ranges_pair, bounds};
164}

166void GLSceneViewport::InitHeatmapMaterials() {
167 auto heatmaps = scene_->heatmaps();
168 assert (heatmaps.size());
169 heatmap_materials_ = std::vector<HeatmapMaterial>();
170 heatmap_materials_->reserve(heatmaps.size() - 1);
171 if (heatmaps.size() == 1) {
172 const auto& heatmap = heatmaps.front();
173 heatmap_materials_->emplace_back(
174 CreateMaterial(heatmap, heatmap, scene_->bounds()));
175 }
176 for (size_t i = 0; i < heatmaps.size() - 1; ++i) {
177 const auto& bot_heatmap = heatmaps[i];
178 const auto& top_heatmap = heatmaps[i + 1];
179 heatmap_materials_->emplace_back(
180 CreateMaterial(bot_heatmap, top_heatmap, scene_->bounds()));
181 }
182
183 // TODO: move this to more appropriate place
184 glm::vec3 center_of_bounds = {scene_->bounds().first, scene_->bounds().second,
185 0.0f};
186 center_of_bounds /= 2;
187 camera_controller_->SetPosition(center_of_bounds);
188}

212void GLSceneViewport::MoveCamera(const Vec2D /*screenPoint*/,
213 const Vec2D delta) {
214 static constexpr float zoom_coefficient = 0.002f;
215 const float zoom_level = camera_controller_->camera().zoom_level();
216
217 const float x_coefficient = -zoom_level * zoom_coefficient;
218 const float y_coefficient = zoom_level * zoom_coefficient;
219
220 const auto dx = static_cast<float>(delta.x);
221 const auto dy = static_cast<float>(delta.y);
222
223 const auto horizontal_translation =
224 x_coefficient * dx * camera_controller_->camera().rightVector();
225 const auto vertical_translation =
226 y_coefficient * dy * camera_controller_->camera().upVector();
227
228 camera_controller_->Translate(horizontal_translation + vertical_translation);
229}

231void GLSceneViewport::RotateCamera(const Vec2D /*screenPoint*/,
232 const Vec2D delta) {
233 static constexpr float zoom_coefficient = 1.0f / 500.0f;
234 const float zoom_level = 1.0f; // camera_controller_->camera().zoom_level();
235
236 const float y_coefficient = zoom_level * zoom_coefficient;
237 const float x_coefficient = -zoom_level * zoom_coefficient;
238
239 const auto dx = static_cast<float>(delta.x);
240 const auto dy = static_cast<float>(delta.y);
241
242 camera_controller_->AddLongitude(x_coefficient * dx);
243 camera_controller_->AddLatitude(y_coefficient * dy);
244}
